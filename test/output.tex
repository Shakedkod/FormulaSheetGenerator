\section{מכניקה קלאסית - דף נוסחאות}
\subsection{קינמטיקה}
מהירות ותאוצה:
\[
v = \frac{dx}{dt}, \quad a = \frac{dv}{dt}
\]
תנועה באצה קבועה:
\[
x = x_0 + v_0 t + \frac{1}{2}at^2
\]
\[
v^2 = v_0^2 + 2a(x - x_0)
\]

\subsubsection{זריקה אופקית}
מרחק נפילה:
\[
x(t_1) = x_0 + \frac{v_0^2 \sin 2\theta}{g}
\]

\subsection{דינמיקה}
חוקי ניוטון:
\begin{itemize}
\item $\sum \vec{F}_i = 0$ — הגוף יתמיד בתנועתו
\item $\sum \vec{F}_i = \frac{d\vec{P}}{dt}$ — חוק שני
\item $\vec{F}_{1,2} = -\vec{F}_{2,1}$ — חוק שלישי
\end{itemize}
כוח חיכוך:
\[
|\vec{F}_{max}| = \mu_s \cdot N, \quad |\vec{F}_\mu| = \mu_k \cdot N
\]

\subsection{וקטורים}
גודל וקטור:
\[
|\vec{A}| = \sqrt{A_x^2 + A_y^2 + A_z^2}
\]

מכפלה סקלרית:
\[
\vec{A} \cdot \vec{B} = |\vec{A}||\vec{B}|\cos\theta
\]

מכפלה וקטורית:
\[
\vec{A} \times \vec{B} = |\vec{A}||\vec{B}|\sin\theta \cdot \hat{n}
\]

\subsection{עבודה ואנרגיה}
אנרגיה קינטית:
\[
E_k = \frac{1}{2}mv^2
\]

משפט עבודה-אנרגיה:
\[
\sum \vec{F} \cdot \Delta\vec{r} = \Delta E_k
\]

אנרגיה מכנית: $E_m = E_k + U$ \\ 
כוח שעבודתו לא תלויה במסלול נקרא \textbf{משמר}. לכל כוח משמר $\nabla \times \vec{F} = 0$.

\subsection{תנועה מעגלית}
מהירות זוויתית: $\omega = \dot{\theta}$ \\ 
תאוצה צנטריפטלית:
\[
a_c = \frac{v^2}{R} = \omega^2 R
\]

\subsection{תנע זוויתי}
מומנט כוח: $\vec{\tau} = \vec{r} \times \vec{F}$ \\ 
תנע זוויתי: $\vec{J} = \vec{r} \times \vec{P}$ \\ 
שימור תנע זוויתי:
\[
\sum \vec{\tau}_i = 0 \implies \vec{J} = \text{const}
\]

מומנט אינרציה:
\[
I = \int r^2 \, dm
\]

\subsection{יחסות פרטית}

פרמטר לורנץ:

$$\gamma = \frac{1}{\sqrt{1 - v^2/c^2}}$$

התקצרות אורך:

$$L = \frac{L_0}{\gamma}$$
